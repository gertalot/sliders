# Slider Hooks

## What is this?

This package provides a number of [React Custom Hooks](https://react.dev/learn/reusing-logic-with-custom-hooks)
that make it easy to build React components such as sliders and dials.

## Demo

```tsx live component=UseDragToMoveSVGDemo
import { useEffect, useRef, useState } from "react";
import { useRotateToMove, Point2D, angleToPoint, pointEquals } from "@gertalot/sliders";

const UseDragToMoveSVGDemo = () => {
  const containerRef = useRef<SVGSVGElement>(null);
  const targetRef = useRef<SVGCircleElement>(null);
  const center = { x: 0, y: 0 };
  const radius = 0.8;

  const { isRotating, isOnTarget, angle, totalAngle, fullRotations } = useRotateToMove({
    containerRef,
    targetRef,
  });

  const [position, setPosition] = useState<Point2D | null>(null);

  useEffect(() => {
    const newPosition = angleToPoint(angle, center, radius);
    setPosition((prev) => (pointEquals(prev, newPosition) ? prev : newPosition));
  }, [angle]);

  return (
    <>
      <svg className="w-full h-full stroke-red-900" viewBox="-1 -1 2 2" ref={containerRef}>
        <line
          x1={center.x}
          y1={center.y}
          x2={position?.x}
          y2={position?.y}
          strokeWidth={radius / 20}
          strokeLinecap="round"
        />
        <circle
          ref={targetRef}
          className="cursor-pointer"
          cx={position?.x}
          cy={position?.y}
          r={radius / 8}
          fill="white"
          strokeWidth={radius / 20}
        />
      </svg>
      <div className=" inset-0 flex flex-col items-start">
        <div className="text-xs text-mono">
          <div>Drag the target, use the mouse wheel, or drag vertically</div>

          <div>
            angle: {totalAngle.toFixed(2)} ({fullRotations} full rotations)
          </div>
          <div>{[isOnTarget ? "on target" : "", isRotating ? "rotating" : ""].filter(Boolean).join(", ")}</div>
        </div>
      </div>
    </>
  );
};

export default UseDragToMoveSVGDemo;
```

## When should I use this?

This package is useful if you want to create React components that respond to the user dragging or scrolling,
such as sliders (e.g. a volume slider), or dials (e.g. a thermostat).
